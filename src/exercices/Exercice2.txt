enum EGeometryType {
  CIRCLE = "circle",
  POLYGON = "polygon",
  POINT = "point",
}

enum EAltitudeRef {
  AGL = "AGL",
  AMSL = "AMSL",
  FL = "AGL",
}

interface IGeoPoint {
  longitude: number;
  latitude: number;
}

interface IGeoPoint3D extends IGeoPoint {
  altitude: number;
  altitudeRef?: EAltitudeRef;
}

interface ICircle {
  center: IGeoPoint;
  radius: number;
}

interface IPolygon {
  points: IGeoPoint[];
}

interface IGeometry {
  geometryType: EGeometryType;
  geometry: ICircle | IGeoPoint | IPolygon;
}

interface IArea extends IGeometry {
  color?: string;
  name? : string;
}
